\subsection*{Questions}
\begin{questions}
  \question Compute the following indefinite integrals.
    \begin{parts}
      \part $ \rint x \cos 5x \dif{x} $
      \part $ \rint \cos x \ln \sin x \dif{x} $
      \part $ \rint \cos \sqrt{x} \dif{x} $
    \end{parts}
  \question Evaluate:
    \begin{parts}
      \part $ \displaystyle\rint^{\sqrt{\pi}}_{\sqrt{\pi/2}} \theta^3 \cos (\theta^2) \dif{\theta} $
      \part $ \displaystyle\rint^1_0 (x^2 + 1)e^{-x} \dif{x} $
    \end{parts}
  \question
    \begin{parts}
      \part Prove that $ \rint (\ln x)^n \dif{x} = x(\ln x)^n - \rint (\ln x)^{(n-1)} \dif{x} $.
      \part Find $ \rint (\ln x)^3 \dif{x} $.
    \end{parts}
\end{questions}
