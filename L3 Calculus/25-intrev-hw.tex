\subsection*{Questions}
\begin{questions}
  \question Compute the following integrals.
    \begin{parts}
      \part $ \rint^2_1 \sin x \dif{x} $
      \part $ \rint \frac{u^2 + 1}{u^3 + 3u} \dif{u} $
      \part $ \rint^{\pi/6}_0 \tan x \dif{x} $
    \end{parts}
  \question Suppose $ y'(x) = \frac{3x^2 + 4x - 4}{2y(x) - 4} $ and $ y(1) = 3 $. Find $ y(x) $ explicitly.
  \question Let $ \omega(a, x) = \rint^x_0 \frac{a^3}{t^2 + a^2} \dif{t} $.
    \begin{parts}
      \part Find $ \omega(a, x) $ explicitly in terms of $ a $ and $ x $. You may wish to use the substitution $ t = a\tan \theta $.
      \part Compute $ \omega(2, 2) $ exactly.
      \part Find $ x $ such that $ \omega(\sqrt{3}, x) = \pi $.
    \end{parts}
  \question Scholarship 2016: Compute the following integral, giving your answer in exact form.
    \begin{displaymath}
      \rint_{-\pi/2}^{\pi/2} (\sin^5 x + \cos^5 x) \dif{x}
    \end{displaymath}
\end{questions}
