\begin{questions}
  \question Find the derivatives with respect to $ x $:
    \begin{parts}
      \part $ y = \tan^{-1} (x^2) $
      \part $ \tan f(x) = x $
      \part $ g(x) = \arctan(\arcsin \sqrt{x}) $
    \end{parts}
  \question Show that
    \begin{displaymath}
      \od{}{x} \left( \frac{1}{2} \tan^{-1} x + \frac{1}{4} \ln\frac{(x + 1)^2}{x^2 + 1} \right) = \frac{1}{(1 + x)(1 + x^2)}
    \end{displaymath}
  \question Prove that $ \od{}{x} \cot^{-1} x = -\frac{1}{x^2 + 1} $.
\end{questions}
