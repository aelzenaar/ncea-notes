\section*{Calculus of Happiness}

\titledquestion{Equations of Fun}
  Take the following system of equations:

  \begin{align*}
    9^{2x+y} - 9^x \times 3^y &= 6\\
    \log_{x+1}(y+3) + \log_{x+1}(y+x+4) &= 3
  \end{align*}

  \begin{parts}
    \part
    \label{part:FirstEquation}
    \begin{subparts}
      \subpart
        Express $ 9^{2x+y} - 9^x \times 3^y = 6 $ in the form $ u^2 - u - 6 = 0 $
        where $ u=3^{2x+y} $.
      \subpart
        Solve the quadratic equation for $ u $, and hence find possible real values
        for $ 2x + y $.
    \end{subparts}

    \part
    \begin{subparts}
      \subpart
        Express the relation $ \log_{x+1}(y+3) + \log_{x+1}(y+x+4) = 3 $ in the form
        $ p(x,y) = (x+1)^3 $ for some polynomial $ p $.
      \subpart
        Expand the new equation into the form $ q(x,y) = 0 $ for some other
        polynomial $ q $.
      \subpart
        Substitute in the expression linking $ x $ to $ y $ from part (\ref{part:FirstEquation})
        above, and hence solve the system of simultaneous equations for real values of $ x $ and $ y $.
    \end{subparts}
  \end{parts}

\titledquestion{Parabola of Joy}
  A car is driving along a road shaped like a parabola at night. The parabola has a vertex at the origin,
  and the car starts at a point 100 m west and 100 m north of the origin.
  \begin{parts}
    \part
      Write an equation modelling the road as a parabola.
    \part
      Find the general equation for the tangent line to the parabola at some point $ (x_0, y_0) $, and substitute into it
      the parabola equation to obtain an equation only in $ x $, $ x_0 $, and $ y_0 $.
    \part
      Suppose there is a statue of the Roman emperor Augustus located 100 m east and 50 m north of the origin. Write the
      equation for the tangent line of the parabola passing through the statue (so that it only depends on a value $ x $
      on the parabola).
    \part
      Solve this equation for $ x $, and hence find the single point $ (x,y) $ on the road where the headlights of the car
      illuminate the statue.
  \end{parts}

\titledquestion{Limits of Excitement}
  A function $ f(x) $ is \emph{continuous} at some point $ (x_0, f(x_0)) $ if and only if
  $ \lim_{x \to x_0} f(x) $ exists and is equal to $ f(x_0) $.
  \begin{parts}
    \part
      Find a value of $ k $ such that $ F(x) $ is continuous at $ x = -3 $ where
      \begin{equation*}
        F(x) =
        \begin{cases}
          \frac{x^2 - 9}{x+3} & \text{if } x \neq -3\\
          k                   & \text{if } x = -3
        \end{cases}
      \end{equation*}
    \part
      Show whether or not $ g(x) $ is continuous at 2, 3, and 4 where
      \begin{equation*}
        g(x) =
        \begin{cases}
          2x-x^2              & \text{if } 0 \leq 2\\
          2-x                 & \text{if } 2 < x \leq 3\\
          x-4                 & \text{if } 3 < x \leq 4\\
          \pi                 & \text{if } x \geq 4
        \end{cases}
      \end{equation*}
    \part
      Find all values of $ \alpha $ such that $ \Phi $ is continuous on $ \mathbb{R} $ where
      \begin{equation*}
        \Phi(x) =
        \begin{cases}
          x+1                 & \text{if } x \leq \alpha \\
          x^2                 & \text{if } x > \alpha
        \end{cases}
      \end{equation*}
  \end{parts}
