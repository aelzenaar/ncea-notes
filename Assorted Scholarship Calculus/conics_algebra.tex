\section*{Scholarship Calculus}

\titledquestion{Conics of Clarity}
  Consider the two conics:
  \begin{align*}
    x^2 - y^2 + 4y - 20 &= 0\\
    x^2 - 8x - 8y^2 &= 0
  \end{align*}

  \begin{parts}
    \part
      Classify and describe both conics.

    \part
      Find the point(s) of intersection of the two conics.
  \end{parts}

\titledquestion{Polynomials of Pleasure}
  Consider the equation $ x^5 = 1 $. One of the solutions is obviously $ x = 1 $; the
  other four solutions are complex.

  \begin{parts}
    \part
      Draw all five solutions to this equation on an Argand diagram, labelling the
      angles between them (and label the non-real roots).

    \part
      Given that $ x - 1 $ is a factor of the equation, show that for any complex root
      $ \alpha $ such that $ \alpha^5 = 1 $ then $ \alpha^4 + \alpha^3 + \alpha^2 + \alpha = -1 $.

    \part
    \begin{subparts}
      \subpart
        Show that the minimal polynomial with roots $\alpha^2 + \alpha^3 $ and $ \alpha + \alpha^4 $ is equivalent to
        $ p(x) = x^2 - x(\alpha + \alpha^2 + \alpha^3 + \alpha^4) + (\alpha + \alpha^4)(\alpha^2 + \alpha^3) $.

      \subpart
        Simplify $ p(x) $ to obtain a polynomial with real coefficients.
    \end{subparts}

  \end{parts}

\titledquestion{Multiplying Midgets}
  We can model a simple population (of animals, fungi, bacteria, and so on) using the \emph{logistic equation}. The logistic equation is given by

  \begin{equation*}
    \frac{dy}{dx} = ky(1-\frac{y}{L})
  \end{equation*}

  where $ k $ and $ L $ are constants. $ L $ is called the carring capacity of the population.

  \begin{parts}
    \part
      Describe the behaviour of a population as $ y $ tends to $ L $ (without solving the differential equation). Does this behaviour make sense
      in the context of a population? Why might $ L $ be called the "carrying capacity"?

    \part
    \begin{subparts}
      \subpart
        Find the general solution of the differential equation for $ y $.

      \subpart
        Find the particular solution where $ L = 5 $, $ k = 2 $, and $ y(0) = 8 $.
    \end{subparts}
  \end{parts}
